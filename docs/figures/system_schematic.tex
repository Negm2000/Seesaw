\documentclass[tikz, border=15pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta, calc, patterns, decorations.pathreplacing}
\usepackage{amsmath, lmodern}
\usepackage[T1]{fontenc}

\begin{document}
\begin{tikzpicture}[>=Stealth, thick]

  \def\tilt{11}
  \def\halflen{5.2}
  \def\trackY{0.55}  %% Distance from pivot to track surface
  \def\cartH{0.8}    %% Height of cart

  %%──────────────────────────────────────────────────────────────────────────
  %%  GROUND + PIVOT SUPPORT
  %%──────────────────────────────────────────────────────────────────────────
  \draw[thick] (-3.0, -1.6) -- (3.0, -1.6);
  \fill[pattern=north east lines, pattern color=gray!40]
    (-3.0, -1.85) rectangle (3.0, -1.6);

  %% Triangle pivot support
  \draw[fill=gray!18, thick] (-0.52, -1.6) -- (0.52, -1.6) -- (0, 0) -- cycle;
  \filldraw (0, 0) circle (3pt);

  %%──────────────────────────────────────────────────────────────────────────
  %%  BEAM + CART  (drawn in rotated frame)
  %%──────────────────────────────────────────────────────────────────────────
  \begin{scope}[rotate=\tilt]

    %% Bracket connecting beam to pivot (fills the visual gap)
    \draw[fill=gray!30, thick] (-0.2, 0) -- (-0.2, {\trackY-0.4}) -- (0.2, {\trackY-0.4}) -- (0.2, 0) -- cycle;

    %% Beam (seesaw plank)
    \draw[fill=gray!7, thick] (-\halflen, {\trackY-0.4}) rectangle (\halflen, \trackY);
    %% Plank label (centered vertically inside the beam)
    \node[font=\small \itshape, rotate=\tilt, gray!70] at (-2.5, {\trackY-0.2}) {Seesaw plank};

    %% Track surface
    \draw[blue!25, thick] ({-\halflen+0.2}, \trackY) -- ({\halflen-0.2}, \trackY);

    %% Cart body
    \pgfmathsetmacro{\cartTY}{\trackY+\cartH}
    \draw[fill=blue!6, thick] (0.8, \trackY) rectangle (2.7, \cartTY);

    %% Motor symbol
    \draw[fill=white, thick] (1.15, {\trackY+0.3}) circle (0.18);
    \node[font=\tiny, rotate=\tilt] at (1.15, {\trackY+0.3}) {M};

    %% Cart label
    \node[font=\small, rotate=\tilt] at (2.0, {\trackY+0.4}) {$M_c$};

    %% Rack (dashed line along track)
    \draw[gray!40, densely dashed] ({-\halflen+0.5}, {\trackY+0.05}) -- ({\halflen-0.5}, {\trackY+0.05});

    %% Cart travel arrows (local motion reference)
    \draw[<->, blue!20, semithick] (0.8, {\cartTY+0.08}) -- (2.7, {\cartTY+0.08});

    %% Center reference dashed lines for dimensions
    \draw[gray!70, densely dashed] (0, 0) -- (0, {\cartTY+0.6});
    \coordinate (CartCoG) at (1.75, {\trackY + \cartH/2});
    \draw[gray!70, densely dashed] (1.75, {\trackY+\cartH/2}) -- (1.75, {\cartTY+0.6});

    %% x_c dimension (pivot to cart centre, moved ABOVE the cart)
    \draw[<->, red!65, semithick]
      (0, {\cartTY+0.4}) -- node[above, font=\small, rotate=\tilt]{$x_c$} (1.75, {\cartTY+0.4});

    %% End-stop marks
    \draw[red!35, thick] ({-\halflen+0.2}, \trackY) -- ({-\halflen+0.2}, {\trackY+0.45});
    \draw[red!35, thick] ({\halflen-0.2},  \trackY) -- ({\halflen-0.2},  {\trackY+0.45});

    %% CoG symbol (Cart Center)
    \filldraw[red!80] (CartCoG) circle (1.5pt);

    %% Dimensions D_T and D_C
    \draw[<->, semithick] (-1.0, 0) -- node[left, font=\small, rotate=\tilt]{$D_T$} (-1.0, \trackY);
    \draw[<->, semithick] (3.1, \trackY) -- node[right, font=\small, rotate=\tilt]{$D_C$} (3.1, {\trackY + \cartH/2});

  \end{scope}

  %%──────────────────────────────────────────────────────────────────────────
  %%  ANGLE ARC  (global frame)
  %%──────────────────────────────────────────────────────────────────────────
  %% Horizontal reference
  \draw[gray!45, dashed] (0, 0) -- ({\halflen+0.8}, 0);

  %% Arc for alpha
  \draw[->, semithick] (4.2, 0) arc[start angle=0, end angle=\tilt, radius=4.2];
  \node[font=\small] at ({4.6*cos(\tilt/2)}, {4.6*sin(\tilt/2)}) {$\alpha$};

  %%──────────────────────────────────────────────────────────────────────────
  %%  GRAVITY ARROW
  %%──────────────────────────────────────────────────────────────────────────
  \draw[->, very thick, red!55] (CartCoG) -- ++(0, -1.35) node[right, font=\small]{$M_c\vec{g}$};

  %%──────────────────────────────────────────────────────────────────────────
  %%  PARAMETER SUMMARY (bottom)
  %%──────────────────────────────────────────────────────────────────────────
  \node[font=\small, anchor=north, text width=14cm, align=center] at (0, -2.3)
    {$M_{\mathrm{SW}} = 3.6\;\mathrm{kg}$\qquad
     $M_c = 0.38\;\mathrm{kg}$\qquad
     $D_T = 0.125\;\mathrm{m}$\qquad
     $D_C = 0.058\;\mathrm{m}$\qquad
     $\alpha_{\max} = \pm 11.5^\circ$};

\end{tikzpicture}
\end{document}
